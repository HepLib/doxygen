230 ex exL=0, exR=0;
231 map<ex,int,ex_is_less> flags;
232 lst exRs;
233 for(auto intg : Integrals) flags[F(ProblemNumber,intg)] = 1;
234 Rules.remove_all();
235 for(auto line : strvec) {
236 if(line.size()==0) {
237 if(!is_zero(exL)) {
238 exR = exR.subs(bMAP);
239 Rules.append(exL==exR);
240 flags[exL] = 0;
241 exRs.append(exR);
242 }
243 exL = exR = 0;
244 } else if(is_zero(exL)) {
245 exL -= Fin(line);
246 if(!exL.match(F(w1,w2))) {
247 cout << line << endl;
248 throw Error("KIRA::Import error found.");
249 }
250 } else {
251 exR += Fin(line);
252 }
253 }
254 if(!is_zero(exL)) {
255 exR = exR.subs(bMAP);
256 Rules.append(exL==exR);
257 flags[exL] = 0;
258 exRs.append(exR);
259 }
260 MIntegrals.remove_all();
261 for(auto kv : flags) {
262 if(kv.second!=0) MIntegrals.append(kv.first);
263 }
264 exset miset;
265 find(exRs,F(w1,w2),miset);
266 for(auto mi : miset) MIntegrals.append(mi);
267 MIntegrals.sort();
268 MIntegrals.unique();
