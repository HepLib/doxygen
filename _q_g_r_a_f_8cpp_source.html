128 lines.append(lst{ e.op(1), iWF(e.op(1)), lst{e.op(0)}, e.op(2) });
129 } else if(isFunction(e,"InField")) {
130 lines.append(lst{ iWF(e.op(1)), e.op(1), lst{e.op(0)}, e.op(2) });

132 lines.append(lst{ iWF(e.op(0).op(1)), iWF(e.op(1).op(1)), lst{e.op(0).op(0), e.op(1).op(0)}, e.op(2) });

135 cid++;
136 v2id[e]=cid;
137 lst fs;
138 for(auto f : e) {
139 ex fid = f.op(1);
140 fid2vid[fid] = cid;
141 fs.append(f.op(0));
142 }
143 vid2fs[cid] = fs;
144 }
145 }
146 }

283 vector<lst> ShrinkCut(ex amp, lst prop, int n) {
284 vector<lst> ret;
285 auto tls = TopoLines(amp);
286 vector<int> cls_vec;
287 for(int i=0; i<tls.nops(); i++) {
288 auto pi = tls.op(i).op(2);
289 if(pi.nops()<2) continue;
290 if(is_zero(pi.op(0)-prop.op(0)) && is_zero(pi.op(1)-prop.op(1))) cls_vec.push_back(i);
291 }
292 if(cls_vec.size()<n) return ret;
293
294 Combinations(cls_vec.size(), n, [n,&ret,cls_vec,tls](const int * is)->void {
295 int cls[n];
296 for(int i=0; i<n; i++) cls[i] = cls_vec[is[i]];;
297
298 // cut each line into 2 half-lines, labeled with 0
299 auto tls2 = tls;
300 for(auto ci : cls) {
301 auto ol = tls2.op(ci);
302 tls2.let_op(ci) = lst{ol.op(0), 0, lst{ol.op(2).op(0)}, ol.op(3)};
303 tls2.append(lst{0, ol.op(1), lst{ol.op(2).op(1)}, ol.op(3)} );
304 }
305
306 // shrink internal lines
307 int last = 0;
308 int ntls2 = tls2.nops();
309 while(true) {
310 ex lp = 0;
311 for(int i=last; i<ntls2; i++) {
312 auto li = tls2.op(i);
313 if(is_zero(li) || li.op(2).nops()<2) continue;
314 last = i;
315 lp = li;
316 tls2.let_op(last) = 0;
317 break;
318 }
319 if(is_zero(lp)) break;
320
321 for(int i=0; i<ntls2; i++) {
322 if(is_zero(tls2.op(i))) continue;
323 if(is_zero(tls2.op(i).op(0)-lp.op(0))) tls2.let_op(i).let_op(0) = lp.op(1);
324 if(is_zero(tls2.op(i).op(1)-lp.op(0))) tls2.let_op(i).let_op(1) = lp.op(1);
325 }
326 }
327
328 // final connected parts
329 map<int, lst> con_map;
330 for(auto li : tls2) {
331 if(is_zero(li)) continue;
332 ex key, val;
333 ex fiL = li.op(0);
334 if(is_a<lst>(fiL)) fiL = fiL.op(0);
335 ex fiR = li.op(1);
336 if(is_a<lst>(fiR)) fiR = fiR.op(0);
337 if(fiL>0 && fiR<0) {
338 con_map[ex_to<numeric>(fiL).to_int()].append(fiR);
339 } else if(fiR>0 && fiL<0) {
340 con_map[ex_to<numeric>(fiR).to_int()].append(fiL);
341 } else {
342 if(fiL>0 && is_zero(fiR)) key = fiL;
343 else if(fiR>0 && is_zero(fiL)) key = fiR;
344 else throw Error("ShrinkCut: unexpcected point reached.");
345 val = li.op(2).op(0);
346 con_map[ex_to<numeric>(key).to_int()].append(val);
347 }
348 }
349
350 lst item;
351 for(auto kv : con_map) item.append(kv.second.sort());
352 ret.push_back(item);
353 });
354
355 return ret;
356 }

364 bool HasLoop(ex amp, lst prop) {
365 auto tls = TopoLines(amp);
366 int ntls = tls.nops();
367 // shrink internal lines of prop
368 int last = 0;
369 while(true) {
370 ex lp = 0;
371 for(int i=last; i<ntls; i++) {
372 auto li = tls.op(i);
373 if(is_zero(li) || li.op(2).nops()<2) continue; // external line
374 auto cpi = li.op(2);
375 bool m1 = !(is_zero(cpi.op(0)-prop.op(0)) && is_zero(cpi.op(1)-prop.op(1)));
376 bool m2 = !(is_zero(cpi.op(0)-prop.op(1)) && is_zero(cpi.op(1)-prop.op(0)));
377 if(m1 && m2) continue; // other propagators
378 if(is_zero(li.op(0)-li.op(1))) return true; // a loop found
379 last = i;
380 lp = li;
381 tls.let_op(last) = 0;
382 break;
383 }
384 if(is_zero(lp)) break;
385
386 for(int i=0; i<ntls; i++) {
387 if(is_zero(tls.op(i))) continue;
388 if(is_zero(tls.op(i).op(0)-lp.op(0))) tls.let_op(i).let_op(0) = lp.op(1);
389 if(is_zero(tls.op(i).op(1)-lp.op(0))) tls.let_op(i).let_op(1) = lp.op(1);
390 }
391 }
392 return false;
393 }
