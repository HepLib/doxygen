92 ex VESimplify(ex expr, int epN, int epsN) {
93 auto expr1 = EvalF(expr);
94 if(expr1.has(eps) && !expr1.is_polynomial(eps)) expr1 = series_ex(expr1, eps, epsN);
95 if(expr1.has(ep) && !expr1.is_polynomial(ep)) expr1 = series_ex(expr1, ep, epN);
96 expr1 = collect_ex(expr1, lst{eps,ep});
97 ex ret = 0;
98 for(int si=expr1.ldegree(eps); si<=epsN; si++) {
99 for(int i=expr1.ldegree(ep); i<=epN; i++) {
100
101 auto ccRes = expr1.coeff(eps,si).coeff(ep,i);
102 ccRes = NN(ccRes).expand();
103 if(!is_a<add>(ccRes)) ccRes = lst{ ccRes };
104
105 exmap pvmap;
106 for(auto item : ccRes) {
107 ex cc = 1, cf = 1;
108 if(is_a<mul>(item)) {
109 cc = 1;
110 cf = 1;
111 for(auto ii : item) {
112 if(is_a<numeric>(ii) || ii.match(VE(w1, w2))) {
113 cc *= ii;
114 } else {
115 cf *= ii;
116 }
117 }
118 } else if(is_a<numeric>(item) || item.match(VE(w1, w2))) {
119 cc = item;
120 } else {
121 cf = item;
122 }
123 pvmap[cf] += cc;
124 }
125
126 for(auto kv : pvmap) {
127 auto vf = kv.first;
128 auto cvs = collect_lst(kv.second, VE(w1,w2));
129 ex vIR=0, eI2 = 0, eR2 = 0;
130 for(auto cv : cvs) {
131 auto co = NN(cv.op(0));
132 auto ve = cv.op(1);
133 if(!is_a<numeric>(co)) {
134 cout << cv << endl;
135 throw Error("VESimplify: coefficient of VE is not numeric.");
136 }
137 if(abs(co)>numeric("1E300")) return NaN;
138 vIR += co * ve.op(0);
139 numeric nco = ex_to<numeric>(co);
140 ex ee = ve.op(1) * ve.op(1);
141 eR2 += nco.real_part() * nco.real_part() * ee;
142 eI2 += nco.imag_part() * nco.imag_part() * ee;
143 }
144 if(!is_zero(eR2) || !is_zero(vIR))
145 ret += VE(ex_to<numeric>(vIR).real_part(), sqrt(eR2)) * pow(eps,si) * pow(ep,i) * vf;
146 if(!is_zero(eI2) || !is_zero(vIR))
147 ret += VE(ex_to<numeric>(vIR).imag_part(), sqrt(eI2)) * pow(eps,si) * pow(ep,i) * vf * I;
148 }
149 }}
150
151 return ret.collect(lst{eps,ep}, true);
152 }

164 auto ccRes = expr.expand();

166 if(is_a<add>(ccRes)) {
167 for(auto item : ccRes) ccResList.append(item);
168 } else {
169 ccResList.append(ccRes);

172 ccRes = -100;

174 ex ntmp;
175 if(is_a<mul>(item)) {
176 ntmp = 1;
177 for(auto ii : item) {
178 if(is_a<numeric>(ii) || ii.match(VE(w1, w2))) {

183 ntmp = item;

185 ntmp = NN(abs(ntmp.subs(VE(w1,w2)==w2)));
186 if(ntmp>ccRes) ccRes = ntmp;
187 }

281 if(FT!=NULL) {

285 if(last_ft!=FT) {
286 qREAL x0[xdim];
287 #ifdef _USE_FLOAT128
288 for(int i=0; i<xdim; i++) x0[i]=0.521Q;
289 #else
290 for(int i=0; i<xdim; i++) x0[i]=0.521L;
291 #endif
292
293 #ifdef _USE_FLOAT128
294 qREAL ft0 = fabsq(FT(x0, Parameter));
295 #else
296 qREAL ft0 = fabsl(FT(x0, Parameter));
297 #endif
298 if(ft0<1E-50) ft0 = 1;
299 last_ft = FT;
300 }
301 #ifdef _USE_FLOAT128
302 ft = fabsq(FT(x, Parameter));
303 #else
304 ft = fabsl(FT(x, Parameter));
305 #endif
306 ft = ft/ft0;
307 if(ft<MPFLimit) return 3;
308 else if(ft<QFLimit) return 2;
309 }
310
311 if(xdim <= QXDim || xmin < QXLimit) return 2;
312
313 return 1;
