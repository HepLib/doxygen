123 exmap SortPermutation(const ex & in_expr, const lst & xs) {
124 auto expr = in_expr;
125 bool isPoly = true;
126 exmap xmap; // note that we need xmap.size == xs.nops()
127
128 map<ex,vector<int>,ex_is_less> pgrp;
129 if(!expr.is_polynomial(xs)) {
130 if(Verbose>2) cout << "SortPermutation: NOT a polynomials, try ALL permutations!" << endl;
131 expr = expr.numer_denom();
132 if(true || !expr.op(0).is_polynomial(xs) || !expr.op(1).is_polynomial(xs)) {
133 isPoly = false;
134 for(int i=0; i<xs.nops(); i++) {
135 pgrp[-1].push_back(i); // all permuations
136 xmap[xs.op(i)] = xs.op(i);
137 }
138 } else {
139 expr = expr.op(0) * expr.op(1);
140 }
141 }
142
143 if(isPoly) { // only for polynomials
144 auto cv_lst = collect_lst(expr, xs);
145 exvector cvs;
146 for(auto item : cv_lst) cvs.push_back(item);
147 sort_vec(cvs); // first sort by coefficient
148
149 int nxi = xs.nops();
150 bool first = true;
151 lst xkey[nxi];
152 lst subkey[nxi];
153 ex clast;
154 for(auto cv : cvs) {
155 ex cc = cv.op(0);
156 ex vv = cv.op(1);
157 if(is_zero(cc)) continue;
158 if(!first && !is_zero(cc-clast)) { // equal coefficient to the same key
159 for(int i=0; i<nxi; i++) {
160 sort_lst(subkey[i]);
161 for(auto item : subkey[i]) xkey[i].append(item);
162 subkey[i].remove_all();
163 }
164 }
165 first = false;
166 clast = cc;
167 for(int i=0; i<nxi; i++) subkey[i].append(vv.degree(xs.op(i)));
168 }
169
170 for(int i=0; i<nxi; i++) { // add last subkey
171 sort_lst(subkey[i]); // need sort due to equal coefficient
172 for(auto item : subkey[i]) xkey[i].append(item);
173 subkey[i].remove_all();
174 }
175
176 exvector key_xi;
177 for(int i=0; i<nxi; i++) key_xi.push_back(lst{xkey[i], xs.op(i)});
178 sort_vec(key_xi); // first sort by key
179
180 for(int i=0; i<nxi; i++) {
181 auto xi = key_xi[i].op(1);
182 auto xki = key_xi[i].op(0);
183 xmap[xi] = xs.op(i); // initial x-replacement
184 pgrp[xki].push_back(i);
185 }
186 expr = expr.subs(xmap, nopat); // need to update expr before permutations
187 expr = collect_ex(expr, xs); // need collect here
188 }
189
190 // pgrp - needs to permuation explicitly
191 ex expr_min = expr;
192 exmap xmap_min;
193 long long npt = 1;
194 for(auto pi : pgrp) {
195 int nvi = pi.second.size();
196 for(int i=1; i<=nvi; i++) npt *= i; // nvi!
197 }
198 for(long long np=0; np<npt; np++) {
199 long long npc = np;
200 exmap xmap_tmp;
201 for(auto pi : pgrp) {
202 auto vi = pi.second;
203 int nvi = vi.size();
204 if(nvi<2) continue;
205 long long npti = 1;
206 for(int i=1; i<=nvi; i++) npti *= i; // nvi!
207 int ck = npc % npti; // kth permutation
208 npc = npc / npti;
209
210 // https://stackoverflow.com/questions/1995328/are-there-any-better-methods-to-do-permutation-of-string
211 auto vip = vi;
212 int k = ck;
213 for(int j=1; j<nvi; ++j) { // startoverflow: j starts from 1
214 std::swap(vip[k%(j+1)], vip[j]);
215 k=k/(j+1);
216 }
217
218 for(int j=0; j<nvi; j++) xmap_tmp[xs.op(vi[j])] = xs.op(vip[j]);
219 }
220 ex expr_tmp = expr.subs(xmap_tmp,nopat);
221 if(ex_less(expr_tmp, expr_min)) {
222 expr_min = expr_tmp;
223 xmap_min = xmap_tmp;
224 }
225 }
226
227 for(auto & kv : xmap) kv.second = kv.second.subs(xmap_min,nopat);
228 return xmap;
229 }

237 lst LoopUF(const Base & base, const ex & idx) {

239 auto props = base.Propagators;

241 // handle sign
242 ex sign = 1;
243 for(int i=0; i<props.nops(); i++) {
244 auto ipr = props.op(i);
245
246 if(ipr.has(iEpsilon)) {
247 auto cc = ipr.coeff(iEpsilon);
248 if(is_a<numeric>(cc)) {
249 if(cc<0) { // using +iEpsilon
250 sign = pow(-1, idx.op(i));

262 if(cc<0) { // using +l^2
263 sign = pow(-1, idx.op(i));

328 auto xmap = SortPermutation(uf,xs);
329 ut = (ut.subs(xmap,nopat));
330 ft = (ft.subs(xmap,nopat));
331 return lst{ut, ft, sign};

490 pair<exmap,lst> FindRules(vector<Base*> fs, bool mi, std::function<lst(const Base &, const ex &)> uf) {
491 vector<pair<Base*,ex>> ibp_idx_vec;
492 if(mi) for(auto fi : fs) for(auto item : fi->MIntegrals) ibp_idx_vec.push_back(make_pair(fi, item));
493 else for(auto fi : fs) for(auto item : fi->Integrals) ibp_idx_vec.push_back(make_pair(fi, F(fi->ProblemNumber,item)));
494
495 exvector uf_smi_vec = GiNaC_Parallel(ibp_idx_vec.size(), [&ibp_idx_vec,&uf](int idx)->ex {
496 auto p = ibp_idx_vec[idx];
497 const Base & fi = (*p.first);
498 auto mi = p.second;
499 auto ks = uf(fi,mi.subs(F(w1,w2)==w2));
500 int nk = ks.nops()-1;
501 lst key;
502 for(int i=0; i<nk; i++) key.append(expand(ks.op(i)));
503 lst pi = fi.Propagators;
504 return lst{ key, lst{ pi, ks.op(nk), mi } }; // ks.op(nk) -> the sign
505 }, "FR");
506
507 map<ex,lst,ex_is_less> group;
508 int ntotal = 0;
509 for(auto item : uf_smi_vec) {
510 group[item.op(0)].append(item.op(1));
511 ntotal++;
512 }
513
514 exmap rules;
515 lst int_lst;
516 exset pis;
517 if(true) { // single element case
518 exset ks, vs;
519 for(auto g : group) {
520 if(g.second.nops()==1) {
521 ks.insert(g.first);
522 vs.insert(g.second.op(0));
523 }
524 }
525 for(auto vi : vs) {
526 pis.insert(vi.op(0));
527 int_lst.append(vi.op(2));
528 }
529 for(auto ki : ks) group.erase(ki);
530 }
531
532 while(!group.empty()) {
533 if(pis.size()>0) {
534 exset ks;
535 for(auto g : group) {
536 ex c0, v0;
537 for(auto gi : g.second) {
538 for(auto pi : pis) {
539 if(gi.op(0).is_equal(pi)) {
540 ks.insert(g.first);
541 c0 = gi.op(1);
542 v0 = gi.op(2);
543 goto found;
544 }
545 }
546 }
547 continue; // if not found pi
548 found: ;
549 int_lst.append(v0);
550 for(auto gi : g.second) {
551 auto ci = gi.op(1);
552 auto vi = gi.op(2);
553 if(v0.is_equal(vi)) continue;
554 rules[vi] = v0 * c0 / ci;
555 }
556 }
557 for(auto ki : ks) group.erase(ki);
558 if(group.empty()) break;
559 }
560 pis.clear();
561
562 int cur = 0;
563 for(auto g : group) {
564 cur++;
565 if(cur>100) break;
566 pis.insert(g.second.op(0).op(0));
567 }
568 }
569
570 if(Verbose>2) cout << " \\--FindRules: " << WHITE << ntotal << " :> " << int_lst.nops() << RESET << " @ " << now(false) << endl;
571 return make_pair(rules,int_lst);
572 }
