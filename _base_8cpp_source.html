92 lst SortPermutation(const ex & in_expr, const lst & xs) {
93 #ifdef using_sort_cache
94 map<ex,bool,ex_is_less> sort_cache;
95 #endif
96 auto expr = in_expr;
97 bool isPoly = true;
98 lst xRepl;
99 map<ex,vector<int>,ex_is_less> pgrp;
100 if(!expr.is_polynomial(xs)) {
101 if(Verbose>2) cout << "SortPermutation: NOT a polynomials, try ALL permutations!" << endl;
102 expr = expr.numer_denom();
103 if(true || !expr.op(0).is_polynomial(xs) || !expr.op(1).is_polynomial(xs)) {
104 isPoly = false;
105 xRepl = xs;
106 for(int i=0; i<xs.nops(); i++) pgrp[-1].push_back(i); // all permuations
107 } else {
108 expr = expr.op(0) * expr.op(1);
109 }
110 }
111
112 if(isPoly) { // only for polynomials
113 auto cv_lst = collect_lst(expr, xs);
114 exvector cvs;
115 for(auto item : cv_lst) cvs.push_back(item);
116 #ifdef using_sort_cache
117 sort_vec(cvs,sort_cache);
118 #else
119 sort_vec(cvs);
120 #endif
121
122 int nxi = xs.nops();
123 bool first = true;
124 lst xkey[nxi];
125 lst subkey[nxi];
126 ex clast;
127 for(auto cv : cvs) {
128 ex cc = cv.op(0);
129 ex vv = cv.op(1);
130 if(is_zero(cc)) continue;
131 if(!first && !is_zero(cc-clast)) {
132 for(int i=0; i<nxi; i++) {
133 #ifdef using_sort_cache
134 sort_lst(subkey[i],sort_cache);
135 #else
136 sort_lst(subkey[i]);
137 #endif
138 for(auto item : subkey[i]) xkey[i].append(item);
139 subkey[i].remove_all();
140 }
141 }
142 first = false;
143 clast = cc;
144 for(int i=0; i<nxi; i++) subkey[i].append(vv.degree(xs.op(i)));
145 }
146 for(int i=0; i<nxi; i++) {
147 #ifdef using_sort_cache
148 sort_lst(subkey[i],sort_cache);
149 #else
150 sort_lst(subkey[i]);
151 #endif
152 for(auto item : subkey[i]) xkey[i].append(item);
153 subkey[i].remove_all();
154 }
155
156 exvector key_xi;
157 for(int i=0; i<nxi; i++) {
158 key_xi.push_back(lst{xkey[i], xs.op(i)});
159 pgrp[xkey[i]].push_back(i); // i w.r.t. position of xs
160 }
161 #ifdef using_sort_cache
162 sort_vec(key_xi,sort_cache);
163 #else
164 sort_vec(key_xi);
165 #endif
166
167 xRepl.remove_all();
168 for(auto item : key_xi) xRepl.append(item.op(1));
169 }
170
171 // pgrp - needs to permuation explicitly
172 expr = in_expr;
173 for(auto pi : pgrp) {
174 auto vi = pi.second;
175 int nvi = vi.size();
176 if(nvi<2) continue;
177 ex expr1 = expr;
178 auto xRepl1 = xRepl;
179
180 // https://stackoverflow.com/questions/1995328/are-there-any-better-methods-to-do-permutation-of-string
181 long long nf = 1;
182 for(int i=2; i<=nvi; i++) nf *= i;
183 for(long long ck=1; ck<=nf; ck++) { // n! permutations
184 auto vi2 = vi;
185 int k = ck;
186 for(int j=1; j<nvi; ++j) {
187 std::swap(vi2[k%(j+1)], vi2[j]);
188 k=k/(j+1);
189 }
190
191 exmap x2x;
192 for(int j=0; j<nvi; j++) {
193 if(vi[j]!=vi2[j]) x2x[xs.op(vi[j])]=xs.op(vi2[j]);
194 }
195 ex expr2 = expr.subs(x2x);
196 #ifdef using_sort_cache
197 if(ex_less_cache(expr2,expr1,sort_cache)) {
198 #else
199 if(ex_less(expr2,expr1)) {
200 #endif
201 expr1 = expr2;
202 xRepl1 = ex_to<lst>(xRepl.subs(x2x));
203 }
204 }
205
206 expr = expr1;
207 xRepl = xRepl1;
208 }
209 return xRepl;
210 }

219
220 auto props = base.Propagators;
221
222 // handle sign
223 ex sign = 1;
224 for(int i=0; i<props.nops(); i++) {
225 auto ipr = props.op(i);
226
227 if(ipr.has(iEpsilon)) {
228 auto cc = ipr.coeff(iEpsilon);
229 if(is_a<numeric>(cc)) {
230 if(cc<0) { // using +iEpsilon
231 sign = pow(-1, idx.op(i));
232 props.let_op(i) = ex(0)-props.op(i);
233 }
234 props.let_op(i) = props.op(i).subs(iEpsilon==0);
235 goto sign_done;
236 } else throw Error("LoopUF: sign of iEpsilon NOT determined.");
237 }
238
239 for(auto lp : base.Internal) {
240 if(ipr.degree(lp)==2) {
241 auto cc = ipr.coeff(lp,2);
242 if(is_a<numeric>(cc)) {
243 if(cc<0) { // using +l^2
244 sign = pow(-1, idx.op(i));
245 props.let_op(i) = ex(0)-props.op(i);
246 }
247 goto sign_done;
248 }
249 }
250 }
251
252 sign_done: ;
253 }
254
255 ex ut, ft, uf;
256 lst key;
257 lst xs;
258 exmap x2ax;
259 int nxi=0;
260 int nps = props.nops();
261 ft = 0;
262 for(int i=0; i<nps; i++) {
263 if(is_zero(idx.op(i))) {
264 key.append(0);
265 continue;
266 }
267 key.append(1);
268 if(!is_zero(idx.op(i)-1)) x2ax[x(nxi)] = a(idx.op(i)) * x(nxi);
269 xs.append(x(nxi));
270 ft -= x(nxi) * props.op(i); // only used when no cache item
271 nxi++;
272 }

274 static map<ex,exmap,ex_is_less> cache_by_prop;
275 exmap & cache = cache_by_prop[lst{props,base.Internal}];
276 if(!using_cache || cache.find(key)==cache.end()) { // no cache item
277 ut = 1;
278 ft = expand_ex(ft);
279 ft = subs_all(ft, base.Replacements);
280 for(int i=0; i<base.Internal.nops(); i++) {
281 auto t2 = ft.coeff(base.Internal.op(i),2);
282 auto t1 = ft.coeff(base.Internal.op(i),1);
283 auto t0 = ft.subs(base.Internal.op(i)==0);
284 ut *= t2;
285 if(is_zero(t2)) return lst{0,0,1};
286 ft = exnormal(t0-t1*t1/(4*t2));
287 ft = expand_ex(ft);
288 ft = subs_all(ft, base.Replacements);
289 }
290 ft = exnormal(ut*ft);
291 ft = exnormal(subs_all(ft, base.Replacements));
292 ut = exnormal(subs_all(ut, base.Replacements));
293 uf = exnormal(ut*ft);
294
295 if(using_cache) cache[key] = lst{ut,ft,uf};
296 } else {
297 auto cc = cache[key];
298 ut = cc.op(0);
299 ft = cc.op(1);
300 uf = cc.op(2);
301 }
302
303 ut = ut.subs(x2ax);
304 ft = ft.subs(x2ax);
305 uf = uf.subs(x2ax);
306
307 uf = uf.subs(MapPreSP);
308 auto xRepl = SortPermutation(uf,xs);
309 for(int i=0; i<nxi; i++) xRepl.let_op(i)=(xRepl.op(i)==x(i));
310 ut = (ut.subs(xRepl));
311 ft = (ft.subs(xRepl));
312 return lst{ut, ft, sign};
313 }
