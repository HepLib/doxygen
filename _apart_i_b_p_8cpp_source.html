78 ex F2ex(const ex & expr_in) {
79 ex ret = expr_in;
80 ret = MapFunction([](const ex & e, MapFunction &self)->ex{
81 if(!e.has(F(w1,w2))) return e;
82 else if(e.match(F(w1, w2))) {
83 auto ps = e.op(0);
84 auto ns = e.op(1);
85 ex res = 1;
86 for(int i=0; i<ps.nops(); i++) res *= pow(ps.op(i), ex(0)-ns.op(i));
87 return res;
88 } else return e.map(self);
89 })(ret);
90 return ret;
91 }
