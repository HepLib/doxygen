676 matrix mat(n+2,n);
677 int cc = mat0.cols();
678 if(cc==n) mat=mat0;
679 else {
680 // zero each element
681 for(int r=0; r<n+2; r++) {
682 for(int c=0; c<cc; c++) mat(r,c) = 0;
683 }
684 // n-row from mat0 to mat, note the last 2 rows still 0
685 for(int r=0; r<n; r++) {
686 for(int c=0; c<cc; c++) mat(r,c) = mat0(r,c);
687 }
688 for(int i=0; i<n; i++) {
689 mat(i,cc) = 1;
690 auto r = mat.rank();
691 if(r==n) break;
692 if(r==cc+1) cc++;
693 else mat(i,cc) = 0;
694 }
695 if(mat.rank()!=n) throw Error("ApartIRC failed, NOT full rank.");
696 // last 2 rows from mat0 to mat
697 for(int r=n; r<n+2; r++) {
698 for(int c=0; c<mat0.cols(); c++) mat(r,c) = mat0(r,c);
699 }
700 }
701 return ApartIR(mat, e.op(1));
702 } else return e.map(self);
703 })(expr_in);
