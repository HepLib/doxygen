304  for(auto ci : cls) {

307  tls2.append(lst{0, ol.op(1), lst{ol.op(2).op(1)}, ol.op(3)} );
308  }

313  while(true) {
314  ex lp = 0;

316  auto li = tls2.op(i);
317  if(is_zero(li) || li.op(2).nops()<2) continue;
318  last = i;
319  lp = li;
320  tls2.let_op(last) = 0;
321  break;
322  }
323  if(is_zero(lp)) break;
