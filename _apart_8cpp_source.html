665  else if(e.match(ApartIR(w1,w2))) {

669  for(int r=0; r<n+2; r++) {
670  for(int c=0; c<n; c++) mat(r,c) = 0;
671  }
672  for(int i=0; i<n; i++) mat(i,i) = 1;
673  return ApartIR(mat, e.op(1));
674  }

677  matrix mat(n+2,n);
678  int cc = mat0.cols();
679  if(cc==n) mat=mat0;
680  else {
681  // zero each element
682  for(int r=0; r<n+2; r++) {
683  for(int c=0; c<cc; c++) mat(r,c) = 0;
684  }
685  // n-row from mat0 to mat, note the last 2 rows still 0
686  for(int r=0; r<n; r++) {

688  }
689  for(int i=0; i<n; i++) {
690  mat(i,cc) = 1;

694  else mat(i,cc) = 0;
695  }
696  if(mat.rank()!=n) throw Error("ApartIRC failed, NOT full rank.");
697  // last 2 rows from mat0 to mat
698  for(int r=n; r<n+2; r++) {
699  for(int c=0; c<mat0.cols(); c++) mat(r,c) = mat0(r,c);
700  }
701  }
702  return ApartIR(mat, e.op(1));
703  } else return e.map(self);
