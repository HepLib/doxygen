374expr call(const std::string func, const std::vector<expr> &ev) {
375 int ttl = ev.size();
376 std::ostringstream ss;
377 GiNaC::symtab st;
378 ss << func << "(";
379 for(int i=0; i<ttl; i++) {
380 ss << "p" << i;
381 if(i+1==ttl) ss << ")";
382 else ss << ",";
383 st["p"+std::to_string(i)] = ev[i]._expr;
384 }
385 return expr(HepLib::str2ex(ss.str(), st));
386}
