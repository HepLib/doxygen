1538  bool ex_less(const ex &a, const ex &b) {
1539 
1540  if(a.is_equal(b)) return false;
1541 
1542  // numeric
1543  if(is_a<numeric>(a) && is_a<numeric>(b)) return (evalf(a-b)<0);
1544  if(is_a<numeric>(a)) return true;
1545  if(is_a<numeric>(b)) return false;
1546 
1547  // symbol
1548  if(is_a<symbol>(a) && is_a<symbol>(b)) return (ex2str(a) < ex2str(b));
1549  if(is_a<symbol>(a)) return true;
1550  if(is_a<symbol>(b)) return false;

1552  // matrix
1553  if(is_a<matrix>(a) && is_a<matrix>(b)) {
1554  auto ma = ex_to<matrix>(a);
1555  auto mb = ex_to<matrix>(b);
1556  if(ma.cols() != mb.cols()) return (ma.cols() < mb.cols());
1557  if(ma.rows() != mb.rows()) return (ma.rows() < mb.rows());
1558  for(int c=0; c<ma.cols(); c++) {
1559  for(int r=0; r<ma.rows(); r++) {

1561  return ex_less(ma(r,c),mb(r,c));

1566  if(is_a<matrix>(a)) return false;
