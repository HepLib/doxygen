956 ex series_ex(ex const & expr_in, symbol const &s0, int sn0) {
957 ex expr = expr_in;
958 if(!expr.has(s0)) return expr;
959
960 exset sset;
961 expr.find(pow(s0, w), sset);
962 numeric sn_lcm = 1;
963 exmap repl1st, repl2nd;
964 for(auto pi : sset) {
965 auto sn = expand(pi.op(1));
966 if(!is_a<numeric>(sn)) {
967 ex ex1 = 0, ex2 = 0;
968 for(auto item : add2lst(sn)) {
969 if(!is_a<numeric>(item)) ex1 += item;
970 else ex2 += item;
971 }
972 sn = ex2;
973 static symbol sss;
974 repl1st[pi] = sss * pow(s0, sn);
975 repl2nd[sss] = pow(s0, ex1);
976 }
977 if(!(is_a<numeric>(sn) && ex_to<numeric>(sn).is_rational())) {
978 cerr << "s = " << s0 << endl;
979 cerr << "expr_in = " << expr_in << endl;
980 throw Error("series_ex: Not rational sn = " + ex2str(sn));
981 }
982 sn_lcm = lcm(sn_lcm, ex_to<numeric>(sn).denom());
983 }
984 if(expr.has(sqrt(s0))) sn_lcm = lcm(sn_lcm, numeric(2));
985 expr = expr.subs(repl1st);
986
987 symbol s("s");
988 if(!sn_lcm.is_integer()) throw Error("series_ex: Not integer with " + ex2str(sn_lcm));
989 if(sn_lcm<0) sn_lcm = numeric(0)-sn_lcm;
990 int sn = sn0 * sn_lcm.to_int();
991 expr = expr.subs(pow(s0,w)==pow(s,w*sn_lcm)).subs(sqrt(s0)==pow(s,sn_lcm/2)).subs(s0==pow(s,sn_lcm));
992
993 ex cvs = collect_lst(expr,s);
994 ex ret = 0;
995 for(auto cv : cvs) {
996 bool ok = false;
997 int exN = 1;
998 while(exN<10) {
999 expr = cv.op(1) + pow(s,sn+exN+2)+pow(s,sn+exN+3);
1000 expr = expr.series(s,sn+exN);
1001 ex ot = 0;
1002 for(int i=0; i<expr.nops(); i++) {
1003 if(is_order_function(expr.op(i))) {
1004 ot = expr.op(i);
1005 break;
1006 }
1007 }
1008 if(!is_order_function(ot)) {
1009 cerr << "expr = " << expr << endl;
1010 throw Error("series_ex: Not an Order term with " + ex2str(ot));
1011 }
1012 if(ot.op(0).degree(s)>sn) {
1013 expr = series_to_poly(expr);
1014 expr = collect_ex(expr,s);
1015 for(int i=expr.ldegree(s); (i<=expr.degree(s) && i<=sn); i++) {
1016 ret += cv.op(0) * expr.coeff(s,i) * pow(s0,ex(i)/sn_lcm);
1017 }
1018 ok = true;
1019 break;
1020 }
1021 exN++;
1022 }
1023 if(!ok) throw Error("series_ex seems not working!");
1024 }
1025 ret = ret.subs(s==pow(s0,ex(1)/sn_lcm)); // need this for log-terms
1026 ret = ret.subs(repl2nd);
1027 ret = collect_ex(ret,s0);
1028 return ret;
1029 }

1053 co_epvec_t power_expand_2(const co_epvec_t & co_epv, int n) {
1054 if(n==1) return co_epv;
1055 else if(n==2) {
1056 ex co = pow(co_epv.first,2);
1057 exmap pcmap;
1058 if(true) {
1059 auto epv = co_epv.second;
1060 if(epv.size()>10000) cout << "Warning: power_expand_2: epv.size > 10000" << endl;
1061 for(int i=0; i<epv.size(); i++) {
1062 pcmap[epv[i].first] += 2*co_epv.first*epv[i].second;
1063 pcmap[pow(epv[i].first,2)] += pow(epv[i].second,2);
1064 for(int j=i+1; j<epv.size(); j++) pcmap[epv[i].first*epv[j].first] += 2*epv[i].second*epv[j].second;
1065 }
1066 }
1067 epvec_t epv;
1068 epv.reserve(pcmap.size());
1069 for(auto kv : pcmap) {
1070 if(is_zero(kv.first) || is_zero(kv.second)) continue;
1071 epv.push_back(make_pair(kv.first, kv.second));
1072 }
1073 return make_pair(co, epvec_t(std::move(epv)));
1074 }
1075 int n2 = n/2;
1076 auto co_epv2 = power_expand_2(co_epv, n2);
1077 co_epv2 = power_expand_2(co_epv2, 2);
1078 if((n % 2)==0) return co_epvec_t(std::move(co_epv2));
1079
1080 ex co = co_epv.first * co_epv2.first;
1081 exmap pcmap;
1082 for(auto ep1 : co_epv.second) {
1083 pcmap[ep1.first] += ep1.second * co_epv2.first;
1084 for(auto ep2 : co_epv2.second) pcmap[ep2.first * ep1.first] += ep1.second * ep2.second;
1085 }
1086 for(auto ep2 : co_epv2.second) pcmap[ep2.first] += ep2.second * co_epv.first;
1087
1088 epvec_t epv;
1089 epv.reserve(pcmap.size());
1090 for(auto kv : pcmap) {
1091 if(is_zero(kv.first) || is_zero(kv.second)) continue;
1092 epv.push_back(make_pair(kv.first, kv.second));
1093 }
1094 return make_pair(co, epvec_t(std::move(epv)));
1095 }

1097 pair<ex,epvec_t> inner_expand_collect(ex const &expr_in, std::function<bool(const ex &)> has_func, int depth=0) {
1098 if(!has_func(expr_in)) {
1099 ex co;
1100 epvec_t epv;
1101 co = expr_in;
1102 return make_pair(co,epv);
1103 } if(is_a<add>(expr_in)) {
1104 ex co = 0;
1105 exmap pcmap;
1106 for(auto item : expr_in) {
1107 if(has_func(item)) {
1108 auto co_epv = inner_expand_collect(item, has_func, depth+1);
1109 co += co_epv.first;
1110 for(auto ep : co_epv.second) pcmap[ep.first] += ep.second;
1111 } else co += item;
1112 }
1113 epvec_t epv;
1114 epv.reserve(pcmap.size());
1115 for(auto kv : pcmap) {
1116 if(is_zero(kv.first) || is_zero(kv.second)) continue;
1117 epv.push_back(make_pair(kv.first, kv.second));
1118 }
1119 return make_pair(co, epvec_t(std::move(epv)));
1120 } else if(is_a<mul>(expr_in)) {
1121 ex co = 1;
1122 epvec_t epv;
1123 for(auto item : expr_in) {
1124 if(!has_func(item)) {
1125 for(auto & ep : epv) ep.second *= item;
1126 co *= item;
1127 } else {
1128 exmap pcmap;
1129 auto co_epv = inner_expand_collect(item, has_func, depth+1);
1130 for(auto ep2 : co_epv.second) {
1131 pcmap[ep2.first] += ep2.second * co;
1132 for(auto ep1 : epv) pcmap[ep1.first * ep2.first] += ep1.second * ep2.second;
1133 }
1134 for(auto ep1 : epv) pcmap[ep1.first] += ep1.second * co_epv.first;
1135 co *= co_epv.first;
1136 epv.clear();
1137 epv.reserve(pcmap.size());
1138 for(auto kv : pcmap) {
1139 if(is_zero(kv.first) || is_zero(kv.second)) continue;
1140 epv.push_back(make_pair(kv.first, kv.second));
1141 }
1142 }
1143 }
1144 return make_pair(co,epvec_t(std::move(epv)));
1145 } else if(is_a<power>(expr_in) && expr_in.op(1).info(info_flags::nonnegint)) {
1146 int n = ex_to<numeric>(expr_in.op(1)).to_int();
1147 auto co_epv = inner_expand_collect(expr_in.op(0), has_func, depth+1);
1148 return power_expand_2(co_epv,n);
1149 } else {
1150 ex co = 0;
1151 epvec_t epv;
1152 epv.push_back(make_pair(expr_in, 1));
1153 return make_pair(co,epvec_t(std::move(epv)));
1154 }
1155 throw Error("inner_expand_collect unexpected region reached.");
1156 }

1685 XIntegral::XIntegral(ex loops, ex ps, ex ns) {
1686 ex a = 0;
1687 ex pre = 1;
1688 ex rem = 0;
1689 lst delta;
1690 int xni = 0;
1691 lst funs, exps;
1692 for(int i=0; i<ps.nops(); i++) {
1693 if(is_zero(ns.op(i))) continue;
1694 auto cpi = Symbol::set_all(ps.op(i).expand());
1695 bool ltQ = false;
1696 for(auto li : loops) {
1697 if(cpi.has(li)) {
1698 ltQ = true;
1699 break;
1700 }
1701 }
1702
1703 ex sgn = 0;
1704 if(!ltQ) {
1705 pre *= pow(Symbol::set_all(ps.op(i).expand()), ex(0)-ns.op(i));
1706 ns.let_op(i) = 0;
1707 ps.let_op(i) = 1;
1708 continue;
1709 } else if(is_a<numeric>(ns.op(i)) && (ns.op(i)<0)) {
1710 throw Error("XIntegral, negative powers not supported yet.");
1711 }
1712
1713 // check loop^2
1714 for(auto li : loops) {
1715 if(!is_a<numeric>(cpi.coeff(li,2))) continue; // TODO: make sure positive
1716 numeric nm = ex_to<numeric>(cpi.coeff(li,2));
1717 if(nm.is_zero()) continue;
1718 sgn = nm>0 ? -1 : 1;
1719 break;
1720 }
1721 // check iEpsilon
1722 if(sgn.is_zero()) {
1723 if(!is_a<numeric>(cpi.coeff(iEpsilon))) continue;
1724 numeric nm = ex_to<numeric>(cpi.coeff(iEpsilon));
1725 if(!nm.is_zero()) sgn = nm>0 ? -1 : 1;
1726 }
1727 // others
1728 if(sgn.is_zero()) {
1729 sgn = 1;
1730 if(is_a<numeric>(cpi) && ex_to<numeric>(cpi)>0) sgn = -1;
1731 else sgn = 1;
1732 }
1733
1734 cpi = (ps.op(i)*sgn).subs(iEpsilon==0);
1735 if(sgn==-1) pre *= exp(I * Pi * ns.op(i));
1736 a += ns.op(i);
1737 auto xi = x(xni);
1738 rem += xi * cpi;
1739 delta.append(xi);
1740 if(!is_zero(ns.op(i)-1)) {
1741 funs.append(xi);
1742 exps.append(ns.op(i)-1);
1743 }
1744 pre /= tgamma(ns.op(i));
1745 xni++;
1746 }
1747 rem = Symbol::set_all(rem.expand());
1748 ex dl2 = loops.nops()*(4-2*ep)/2;
1749 pre *= tgamma(a-dl2) * pow(I,loops.nops()) * pow(Pi,dl2) * pow(2*Pi,loops.nops()*(2*ep-4));
1750
1751 // Loop
1752 ex u=1;
1753 for(int i=0; i<loops.nops(); i++) {
1754 ex t2 = rem.coeff(loops.op(i),2);
1755 ex t1 = rem.coeff(loops.op(i),1);
1756 ex t0 = rem.coeff(loops.op(i),0);
1757 u *= (-t2);
1758 if(t2==0) {
1759 Functions = lst{0};
1760 Exponents = lst{1};
1761 return;
1762 }
1763 rem = (t0 - pow(t1,2)/(4*t2)).expand();
1764 }
1765 rem = Symbol::set_all(rem).normal();
1766 u = Symbol::set_all(u).normal();
1767 rem = (rem * u).normal();
1768
1769 funs.prepend(rem);
1770 exps.prepend(dl2-a);
1771 funs.prepend(u);
1772 exps.prepend(a-dl2-(4-2*ep)/2);
1773 auto num_den = numer_denom(pre);
1774 funs.append(num_den.op(0));
1775 exps.append(1);
1776 if(!is_zero(num_den.op(1))) {
1777 funs.append(num_den.op(1));
1778 exps.append(-1);
1779 }
1780
1781 Functions = funs;
1782 Exponents = exps;
1783 Deltas = lst{delta};
1784 }
