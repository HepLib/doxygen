932  typedef vector<pair<ex,ex>> epvec_t; // first-rest, second-coeff
933  pair<ex,epvec_t> inner_expand_collect(ex const &expr_in, std::function<bool(const ex &)> has_func, int depth=0) {
934  if(!has_func(expr_in)) {
935  ex co;
936  epvec_t epv;
937  co = expr_in;
938  return make_pair(co,epv);
939  } if(is_a<add>(expr_in)) {
940  ex co = 0;
941  epvec_t epv;
942  exmap pcmap;
943  for(auto item : expr_in) {
944  if(has_func(item)) {
945  auto co_epv = inner_expand_collect(item, has_func, depth+1);
946  co += co_epv.first;
947  for(auto ep : co_epv.second) pcmap[ep.first] += ep.second;
948  } else co += item;
949  }
950  for(auto kv : pcmap) {
951  if(is_zero(kv.first) || is_zero(kv.second)) continue;
952  epv.push_back(make_pair(kv.first, kv.second));
953  }
954  return make_pair(co,epv);
955  } else if(is_a<mul>(expr_in)) {
956  ex co = 1;
957  epvec_t epv;
958  for(auto item : expr_in) {
959  if(!has_func(item)) {
960  for(auto & ep : epv) ep.second *= item;
961  co *= item;
962  } else {
963  exmap pcmap;
964  auto co_epv = inner_expand_collect(item, has_func, depth+1);
965  for(auto ep2 : co_epv.second) {
966  pcmap[ep2.first] += ep2.second * co;
967  for(auto ep1 : epv) pcmap[ep1.first * ep2.first] += ep1.second * ep2.second;
968  }
969  for(auto ep1 : epv) pcmap[ep1.first] += ep1.second * co_epv.first;
970  co *= co_epv.first;
971  epv.clear();
972  for(auto kv : pcmap) {
973  if(is_zero(kv.first) || is_zero(kv.second)) continue;
974  epv.push_back(make_pair(kv.first, kv.second));
975  }
976  }
977  }
978  return make_pair(co,epv);
979  } else if(is_a<power>(expr_in) && expr_in.op(1).info(info_flags::nonnegint)) {
980  ex co = 1;
981  epvec_t epv;
982  int n = ex_to<numeric>(expr_in.op(1)).to_int();
983  auto co_epv = inner_expand_collect(expr_in.op(0), has_func, depth+1);
984  for(int i=0; i<n; i++) {
985  exmap pcmap;
986  for(auto ep2 : co_epv.second) {
987  pcmap[ep2.first] += ep2.second * co;
988  for(auto ep1 : epv) pcmap[ep1.first * ep2.first] += ep1.second * ep2.second;
989  }
990  for(auto ep1 : epv) pcmap[ep1.first] += ep1.second * co_epv.first;
991  co *= co_epv.first;
992  epv.clear();
993  for(auto kv : pcmap) {
994  if(is_zero(kv.first) || is_zero(kv.second)) continue;
995  epv.push_back(make_pair(kv.first, kv.second));
996  }
997  }
998  return make_pair(co,epv);
999  } else {
1000  ex co = 0;
1001  epvec_t epv;
1002  epv.push_back(make_pair(expr_in, 1));
1003  return make_pair(co,epv);
1004  }
1005  throw Error("inner_expand_collect unexpected region reached.");
1006  }

1471  bool ex_less(const ex &a, const ex &b) {
1472  if(a.is_equal(b)) return false;
1473 
1474  // numeric
1475  if(is_a<numeric>(a) && is_a<numeric>(b)) {
1476  auto ab = a-b;
1477  auto abr = real_part(ab);
1478  if(!is_zero(abr)) return (abr < 0);
1479  else return (imag_part(ab) < 0);
1480  }
1481  if(is_a<numeric>(a)) return true;
1482  if(is_a<numeric>(b)) return false;
1483 
1484  // symbol
1485  if(is_a<symbol>(a) && is_a<symbol>(b)) return (ex2str(a) < ex2str(b));
1486  if(is_a<symbol>(a)) return true;
1487  if(is_a<symbol>(b)) return false;
1488 
1489  // matrix
1490  if(is_a<matrix>(a) && is_a<matrix>(b)) {
1491  auto ma = ex_to<matrix>(a);
1492  auto mb = ex_to<matrix>(b);
1493  if(ma.cols() != mb.cols()) return (ma.cols() < mb.cols());
1494  if(ma.rows() != mb.rows()) return (ma.rows() < mb.rows());
1495  for(int c=0; c<ma.cols(); c++) {
1496  for(int r=0; r<ma.rows(); r++) {
1497  if(ma(r,c).is_equal(mb(r,c))) continue;
1498  return ex_less(ma(r,c),mb(r,c));
1499  }}
1500  return false;
1501  }
1502  if(is_a<matrix>(b)) return true;
1503  if(is_a<matrix>(a)) return false;
1504 
1505  // lst
1506  if(is_a<lst>(a) && is_a<lst>(b)) {
1507  auto na = a.nops();
1508  auto nb = b.nops();
1509  if(na!=nb) return (na<nb);
1510  for(int i=0; i<na; i++) {
1511  if(a.op(i).is_equal(b.op(i))) continue;
1512  return ex_less(a.op(i), b.op(i));
1513  }
1514  return false;
1515  }
1516  if(is_a<lst>(b)) return true;
1517  if(is_a<lst>(a)) return false;
1518 
1519  // atomic
1520  auto an = a.nops();
1521  auto bn = b.nops();
1522  if(an==0 && bn==0) {
1523  string na = a.return_type_tinfo().tinfo->name();
1524  string nb = b.return_type_tinfo().tinfo->name();
1525  auto nc = na.compare(nb);
1526  if(nc<0) return true;
1527  else if(nc>0) return false;
1528  else return (ex2str(a) < ex2str(b));
1529  }
1530 
1531  // power
1532  if(is_a<GiNaC::power>(a) || is_a<GiNaC::power>(b)) {
1533  ex ae=a, be=b;
1534  ex an=1, bn=1;
1535  if(is_a<GiNaC::power>(a)) {
1536  ae = a.op(0);
1537  an = a.op(1);
1538  }
1539  if(is_a<GiNaC::power>(b)) {
1540  be = b.op(0);
1541  bn = b.op(1);
1542  }

1544  if(an.info(info_flags::real) && bn.info(info_flags::real)) return (an<bn);
1545  if(!is_zero(an-bn)) return ex_less(an,bn);

1551  string na = ex_to<GiNaC::function>(a).get_name();
1552  string nb = ex_to<GiNaC::function>(b).get_name();
1553  auto nc = na.compare(nb);
1554  if(nc<0) return true;
1555  else if(nc>0) return false;
1556  if(an!=bn) return (an < bn);
