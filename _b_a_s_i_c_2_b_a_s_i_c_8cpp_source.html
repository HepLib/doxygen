773 ex series_ex(ex const & expr_in, symbol const &s0, int sn0) {
774 ex expr = expr_in;
775 if(!expr.has(s0)) return expr;
776
777 exset sset;
778 expr.find(pow(s0, w), sset);
779 numeric sn_lcm = 1;
780 exmap repl1st, repl2nd;
781 for(auto pi : sset) {
782 auto sn = expand(pi.op(1));
783 if(!is_a<numeric>(sn)) {
784 ex ex1 = 0, ex2 = 0;
785 for(auto item : add2lst(sn)) {
786 if(!is_a<numeric>(item)) ex1 += item;
787 else ex2 += item;
788 }
789 sn = ex2;
790 static symbol sss;
791 repl1st[pi] = sss * pow(s0, sn);
792 repl2nd[sss] = pow(s0, ex1);
793 }
794 if(!(is_a<numeric>(sn) && ex_to<numeric>(sn).is_rational())) {
795 cerr << "s = " << s0 << endl;
796 cerr << "expr_in = " << expr_in << endl;
797 throw Error("series_ex: Not rational sn = " + ex2str(sn));
798 }
799 sn_lcm = lcm(sn_lcm, ex_to<numeric>(sn).denom());
800 }
801 if(expr.has(sqrt(s0))) sn_lcm = lcm(sn_lcm, numeric(2));
802 expr = expr.subs(repl1st);
803
804 static symbol s("s");
805 if(!sn_lcm.is_integer()) throw Error("series_ex: Not integer with " + ex2str(sn_lcm));
806 if(sn_lcm<0) sn_lcm = numeric(0)-sn_lcm;
807 int sn = sn0 * sn_lcm.to_int();
808 expr = expr.subs(pow(s0,w)==pow(s,w*sn_lcm)).subs(sqrt(s0)==pow(s,sn_lcm/2)).subs(s0==pow(s,sn_lcm));
809
810 ex cvs = collect_lst(expr,s);
811 ex ret = 0;
812 for(auto cv : cvs) {
813 bool ok = false;
814 int exN = 1;
815 while(exN<10) {
816 expr = cv.op(1) + pow(s,sn+exN+2)+pow(s,sn+exN+3);
817 expr = expr.series(s,sn+exN);
818 ex ot = 0;
819 for(int i=0; i<expr.nops(); i++) {
820 if(is_order_function(expr.op(i))) {
821 ot = expr.op(i);
822 break;
823 }
824 }
825 if(!is_order_function(ot)) {
826 cerr << "expr = " << expr << endl;
827 throw Error("series_ex: Not an Order term with " + ex2str(ot));

829 if(ot.op(0).degree(s)>sn) {
830 expr = series_to_poly(expr);
831 expr = collect_ex(expr,s);
832 for(int i=expr.ldegree(s); (i<=expr.degree(s) && i<=sn); i++) {
833 ret += cv.op(0) * expr.coeff(s,i) * pow(s0,ex(i)/sn_lcm);
834 }
835 ok = true;
836 break;
837 }
838 exN++;
839 }
840 if(!ok) throw Error("series_ex seems not working!");
841 }
842 ret = ret.subs(s==pow(s0,ex(1)/sn_lcm)); // need this for log-terms
843 ret = ret.subs(repl2nd);
844 ret = collect_ex(ret,s0);
845 return ret;
846 }

927 co *= co_epv.first;
928 epv.clear();
929 for(auto kv : pcmap) {

932 }
933 }
934 return make_pair(co,epv);
935 } else {
936 ex co = 0;
937 epvec_t epv;
938 epv.push_back(make_pair(expr_in, 1));
939 return make_pair(co,epv);
940 }
941 throw Error("inner_expand_collect unexpected region reached.");
942 }

1407 bool ex_less(const ex &a, const ex &b) {
1408 static map<ex,bool,ex_is_less> cache;
1409 ex key = lst{a,b};
1410 if(cache.find(key)!=cache.end()) return cache[key];
1411 if(a.is_equal(b)) return cache[key]=false;
1412
1413 // numeric
1414 if(is_a<numeric>(a) && is_a<numeric>(b)) {
1415 auto ab = a-b;
1416 auto abr = real_part(ab);
1417 if(!is_zero(abr)) return cache[key]=(abr<0);
1418 else return cache[key]=(imag_part(ab)<0);
1419 }
1420 if(is_a<numeric>(a)) return cache[key]=true;
1421 if(is_a<numeric>(b)) return cache[key]=false;
1422
1423 // symbol
1424 if(is_a<symbol>(a) && is_a<symbol>(b)) return cache[key]=(ex2str(a) < ex2str(b));
1425 if(is_a<symbol>(a)) return cache[key]=true;
1426 if(is_a<symbol>(b)) return cache[key]=false;
1427
1428 // matrix
1429 if(is_a<matrix>(a) && is_a<matrix>(b)) {
1430 auto ma = ex_to<matrix>(a);
1431 auto mb = ex_to<matrix>(b);
1432 if(ma.cols() != mb.cols()) return cache[key]=(ma.cols() < mb.cols());
1433 if(ma.rows() != mb.rows()) return cache[key]=(ma.rows() < mb.rows());
1434 for(int c=0; c<ma.cols(); c++) {
1435 for(int r=0; r<ma.rows(); r++) {
1436 if(ma(r,c).is_equal(mb(r,c))) continue;
1437 return cache[key]=ex_less(ma(r,c),mb(r,c));
1438 }}
1439 return cache[key]=false;
1440 }
1441 if(is_a<matrix>(b)) return cache[key]=true;
1442 if(is_a<matrix>(a)) return cache[key]=false;
1443
1444 // lst
1445 if(is_a<lst>(a) && is_a<lst>(b)) {
1446 auto na = a.nops();
1447 auto nb = b.nops();
1448 if(na!=nb) return cache[key]=(na<nb);
1449 for(int i=0; i<na; i++) {
1450 if(a.op(i).is_equal(b.op(i))) continue;
1451 return cache[key]=ex_less(a.op(i), b.op(i));
1452 }
1453 return cache[key]=false;
1454 }
1455 if(is_a<lst>(b)) return cache[key]=true;
1456 if(is_a<lst>(a)) return cache[key]=false;
1457
1458 // atomic
1459 auto an = a.nops();
1460 auto bn = b.nops();
1461 if(an==0 && bn==0) {
1462 string na = a.return_type_tinfo().tinfo->name();
1463 string nb = b.return_type_tinfo().tinfo->name();
1464 auto nc = na.compare(nb);
1465 if(nc<0) return cache[key]=true;
1466 else if(nc>0) return cache[key]=false;
1467 else return cache[key]=(ex2str(a) < ex2str(b));
1468 }
1469
1470 // power
1471 if(is_a<GiNaC::power>(a) || is_a<GiNaC::power>(b)) {
1472 ex ae=a, be=b;
1473 ex an=1, bn=1;
1474 if(is_a<GiNaC::power>(a)) {
1475 ae = a.op(0);
1476 an = a.op(1);
1477 }
1478 if(is_a<GiNaC::power>(b)) {
1479 be = b.op(0);
1480 bn = b.op(1);
1481 }
1482 if(!ae.is_equal(be)) return cache[key]=ex_less(ae,be);
1483 if(an.info(info_flags::real) && bn.info(info_flags::real)) return cache[key]=(an<bn);
1484 if(!is_zero(an-bn)) return cache[key]=ex_less(an,bn);
1485 return cache[key]=false;
1486 }
1487
1488 // function
1489 if(is_a<GiNaC::function>(a) && is_a<GiNaC::function>(b)) {
1490 string na = ex_to<GiNaC::function>(a).get_name();
1491 string nb = ex_to<GiNaC::function>(b).get_name();
1492 auto nc = na.compare(nb);
1493 if(nc<0) return cache[key]=true;
1494 else if(nc>0) return cache[key]=false;
1495 if(an!=bn) return cache[key]=(an < bn);
1496 for(int i=0; i<an; i++) {
1497 if(a.op(i).is_equal(b.op(i))) continue;
1498 return cache[key]=ex_less(a.op(i), b.op(i));
1499 }
1500 return cache[key]=false;
1501 }
1502
1503 // add
1504 if(is_a<add>(a) && is_a<add>(b)) {
1505 auto as = add2lst(a);
1506 auto bs = add2lst(b);
1507 auto na = as.nops();
1508 auto nb = bs.nops();
1509 sort_lst(as,false);
1510 sort_lst(bs,false);
1511 int nn = ((na>nb) ? nb : na);
1512 for(int i=0; i<nn; i++) {
1513 if(as.op(i).is_equal(bs.op(i))) continue;
1514 return cache[key]=ex_less(as.op(i), bs.op(i));
1515 }
1516 if(na!=nb) return cache[key]=(na<nb);
1517 return cache[key]=false;
1518 }
1519 if(is_a<add>(a)) return cache[key]=false;
1520 if(is_a<add>(b)) return cache[key]=true;
1521
1522 // mul
1523 if(is_a<mul>(a) && is_a<mul>(b)) {
1524 auto as = mul2lst(a);
1525 auto bs = mul2lst(b);
1526 auto na = as.nops();
1527 auto nb = bs.nops();
1528 sort_lst(as,false);
1529 sort_lst(bs,false);
1530 int nn = ((na>nb) ? nb : na);
1531 for(int i=0; i<nn; i++) {
1532 if(as.op(i).is_equal(bs.op(i))) continue;
1533 return cache[key]=ex_less(as.op(i), bs.op(i));
1534 }
1535 if(na!=nb) return cache[key]=(na<nb);
1536 return cache[key]=false;
1537 }
1538 if(is_a<mul>(a)) return cache[key]=false;
1539 if(is_a<mul>(b)) return cache[key]=true;
1540
1541 // type
1542 string tna = a.return_type_tinfo().tinfo->name();
1543 string tnb = b.return_type_tinfo().tinfo->name();
1544 auto tnc = tna.compare(tnb);
1545 if(tnc<0) return cache[key]=true;
1546 else if(tnc>0) return cache[key]=false;
1547
1548 // node_number
1549 auto nna = node_number(a);
1550 auto nnb = node_number(b);
1551 if(nna!=nnb) return cache[key]=(nna < nnb);
1552
1553 // all others
1554 if(an!=bn) return (an<bn);
1555 for(int i=0; i<an; i++) {
1556 if(a.op(i).is_equal(b.op(i))) continue;
1557 return cache[key]=ex_less(a.op(i), b.op(i));
1558 }
1559
1560 return cache[key]=(ex2str(a) < ex2str(b));
1561 }

1683 ex a = 0;
1684 ex pre = 1;
1685 ex rem = 0;
1686 lst delta;
1687 int xni = 0;
1688 lst funs, exps;
1689 for(int i=0; i<ps.nops(); i++) {
1690 if(is_zero(ns.op(i))) continue;
1691 auto cpi = Symbol::set_all(ps.op(i).expand());
1692 bool ltQ = false;
1693 for(auto li : loops) {
1694 if(cpi.has(li)) {
1695 ltQ = true;
1696 break;
1697 }
1698 }
1699
1700 ex sgn = 0;
1701 if(!ltQ) {
1702 pre *= pow(Symbol::set_all(ps.op(i).expand()), ex(0)-ns.op(i));
1703 ns.let_op(i) = 0;
1704 ps.let_op(i) = 1;
1705 continue;
1706 } else if(is_a<numeric>(ns.op(i)) && (ns.op(i)<0)) {
1707 throw Error("XIntegral, negative powers not supported yet.");
1708 }
1709
1710 // check loop^2
1711 for(auto li : loops) {
1712 if(!is_a<numeric>(cpi.coeff(li,2))) continue; // TODO: make sure positive
1713 numeric nm = ex_to<numeric>(cpi.coeff(li,2));
1714 if(nm.is_zero()) continue;
1715 sgn = nm>0 ? -1 : 1;
1716 break;
1717 }
1718 // check iEpsilon
1719 if(sgn.is_zero()) {
1720 if(!is_a<numeric>(cpi.coeff(iEpsilon))) continue;
1721 numeric nm = ex_to<numeric>(cpi.coeff(iEpsilon));
1722 if(!nm.is_zero()) sgn = nm>0 ? -1 : 1;
1723 }
1724 // others
1725 if(sgn.is_zero()) {
1726 sgn = 1;
1727 if(is_a<numeric>(cpi) && ex_to<numeric>(cpi)>0) sgn = -1;
1728 else sgn = 1;
1729 }
1730
1731 cpi = (ps.op(i)*sgn).subs(iEpsilon==0);
1732 if(sgn==-1) pre *= exp(I * Pi * ns.op(i));
1733 a += ns.op(i);
1734 auto xi = x(xni);
1735 rem += xi * cpi;
1736 delta.append(xi);
1737 if(!is_zero(ns.op(i)-1)) {
1738 funs.append(xi);
1739 exps.append(ns.op(i)-1);
1740 }
1741 pre /= tgamma(ns.op(i));
1742 xni++;
1743 }
1744 rem = Symbol::set_all(rem.expand());
1745 ex dl2 = loops.nops()*(4-2*ep)/2;
1746 pre *= tgamma(a-dl2) * pow(I,loops.nops()) * pow(Pi,dl2) * pow(2*Pi,loops.nops()*(2*ep-4));
1747
1748 // Loop
1749 ex u=1;
1750 for(int i=0; i<loops.nops(); i++) {
1751 ex t2 = rem.coeff(loops.op(i),2);
1752 ex t1 = rem.coeff(loops.op(i),1);
1753 ex t0 = rem.coeff(loops.op(i),0);
1754 u *= (-t2);
1755 if(t2==0) {
1756 Functions = lst{0};
1757 Exponents = lst{1};
1758 return;
1759 }
1760 rem = (t0 - pow(t1,2)/(4*t2)).expand();
1761 }
1762 rem = Symbol::set_all(rem).normal();
1763 u = Symbol::set_all(u).normal();
1764 rem = (rem * u).normal();
1765
1766 funs.prepend(rem);
1767 exps.prepend(dl2-a);
1768 funs.prepend(u);
1769 exps.prepend(a-dl2-(4-2*ep)/2);
1770 auto num_den = numer_denom(pre);
1771 funs.append(num_den.op(0));
1772 exps.append(1);
1773 if(!is_zero(num_den.op(1))) {
1774 funs.append(num_den.op(1));
1775 exps.append(-1);
1776 }
1777
1778 Functions = funs;
1779 Exponents = exps;
1780 Deltas = lst{delta};
1781 }
