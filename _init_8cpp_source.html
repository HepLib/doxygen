181 MapFunction Rationalize([](const ex & e, MapFunction & self)->ex{
182 if(is_a<numeric>(e)) {
183 auto ne = ex_to<numeric>(e);
184 if(ne.is_crational()) return e;
185 auto zz = ne.to_cl_N();
186 auto re = cln::rationalize(cln::realpart(zz));
187 auto im = cln::rationalize(cln::imagpart(zz));
188 return numeric(cln::complex(re,im));
189 } else return e.map(self);
190 });
